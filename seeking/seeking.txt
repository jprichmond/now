# GENERATE BULLET ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~#
def bullet(s,mx,dent):
  a = []; i = 0
  s = ' '*dent+'* '+s
  while (len(s) > mx):
    i = mx
    while (s[i] != ' '): i -= 1
    a.append(s[:i])
    s = ' '*dent+' '+s[i:]
  a.append(s)
  return a

# GENERATE BULLETS ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~#
def bullets(arr,mx,dent):
  a = []
  for s in arr:
    a.extend(bullet(s,mx,dent))
  return a
